Klik "Selanjutnya" untuk memulai menggunakan komik.

Selanjutnya 
def siapkan_komik(path)
  komik = {}
  File.foreach(path) do |baris|
    nama, url = baris.split(': ')
    komik[nama] = url.strip
  end
  komik
end
